182 template <typename T>
183 bool GetNumericFromBsonIterator(bson_iter_t* iter, T& numericvalue) {
184  const bson_value_t* vv = bson_iter_value(iter);
185  if (vv->value_type == BSON_TYPE_INT32) {
186  numericvalue = static_cast<T>(vv->value.v_int32);
187  } else if (vv->value_type == BSON_TYPE_INT64) {
188  numericvalue = static_cast<T>(vv->value.v_int64);
189  } else if (vv->value_type == BSON_TYPE_DOUBLE) {
190  numericvalue = static_cast<T>(vv->value.v_double);
191  } else if (vv->value_type == BSON_TYPE_UTF8) {
192  string tmp = vv->value.v_utf8.str;
193  if (tmp.find_first_of("0123456789") == string::npos) {
194  cout << "bson iterator isn't or not contains a numeric value." << endl;
195  return false;
196  }
197  char* end = nullptr;
198  numericvalue = static_cast<T>(strtod(tmp.c_str(), &end));
199  } else {
200  cout << "bson iterator isn't or not contains a numeric value." << endl;
201  return false;
202  }
203  return true;
204 }
