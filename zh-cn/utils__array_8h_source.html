245  const INI_T init_value) {
246  if (nullptr != data) {
247  cout << "The input 2D array pointer is not nullptr, without initialized!" << endl;
248  return false;
249  }
250  data = new(nothrow) T *[row];
251  if (nullptr == data) {
252  delete[] data;
253  cout << "Bad memory allocated during 2D array initialization!" << endl;
254  return false;
255  }
256  T init = static_cast<T>(init_value);
257  int bad_alloc = 0;
258 #pragma omp parallel for reduction(+:bad_alloc)
259  for (int i = 0; i < row; i++) {
260  data[i] = new(nothrow) T[col];
261  if (nullptr == data[i]) {
262  delete[] data[i];
263  bad_alloc++;
264  }
265  for (int j = 0; j < col; j++) {
266  data[i][j] = init;
267  }
268  }
269  if (bad_alloc > 0) {
270  cout << "Bad memory allocated during 2D array initialization!" << endl;
271  Release2DArray(row, data);
272  return false;
273  }
274  return true;

309  Release2DArray(row, data);
